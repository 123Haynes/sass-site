#invisible? ⇒ Boolean

def invisible?
  case @type
  when :loud; false
  when :silent; true
  else; style == :compressed
  end
end

def lines
  @value.inject(0) do |s, e|
    next s + e.count("\n") if e.is_a?(String)
    next s
  end
end
