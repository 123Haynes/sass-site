def initialize(build_source_mapping = false)
  @tabs = 0
  @line = 1
  @offset = 1
  @result = String.new("")
  @source_mapping = build_source_mapping ? Sass::Source::Map.new : nil
  @lstrip = nil
  @in_directive = false
end
